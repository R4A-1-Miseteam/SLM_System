function initDB() {
  return { version:'1.1', lastUpdated: new Date().toISOString(), subjects:[], logs:[], todos:[] };
}
function load() {
  try { const r = localStorage.getItem(KEY); return r ? JSON.parse(r) : initDB(); } catch { return initDB(); }
}
function save(d) {
  d.lastUpdated = new Date().toISOString();
  localStorage.setItem(KEY, JSON.stringify(d));
}
function validate(d) {
  return d && typeof d==='object' && d.version && Array.isArray(d.subjects) && Array.isArray(d.logs) && Array.isArray(d.todos);
}
function uid() { return crypto.randomUUID ? crypto.randomUUID() : Math.random().toString(36).slice(2)+Date.now().toString(36); }

let db = load();

// ========== DATE UTILS ==========
function today() { return new Date().toISOString().slice(0,10); }
function getWeek(offset=0) {
  const now=new Date(); now.setDate(now.getDate()+offset*7);
  const d=now.getDay(); const mon=new Date(now); mon.setDate(now.getDate()-(d===0?6:d-1));
  const days=[]; for(let i=0;i<7;i++){const x=new Date(mon);x.setDate(mon.getDate()+i);days.push(x.toISOString().slice(0,10));}
  return days;
}
function getMonthDays() {
  const now=new Date(); const y=now.getFullYear(); const m=now.getMonth();
  const last=new Date(y,m+1,0).getDate();
  const days=[]; for(let i=1;i<=last;i++) days.push(`${y}-${String(m+1).padStart(2,'0')}-${String(i).padStart(2,'0')}`);
  return days;
}

// ========== ACHIEVEMENT ==========
function calcWeek(subject) {
  const days=getWeek();
  const logs=db.logs.filter(l=>l.subjectId===subject.id && days.includes(l.date));
  const totalTime=logs.reduce((a,l)=>a+l.duration,0);
  const totalPage=logs.reduce((a,l)=>a+l.pageCount,0);
  const pctT=subject.targetTime?Math.min(100,Math.round(totalTime/subject.targetTime*100)):0;
  const pctP=subject.targetPage?Math.min(100,Math.round(totalPage/subject.targetPage*100)):0;
  const pct=subject.targetPage?Math.round((pctT+pctP)/2):pctT;
  return {totalTime,totalPage,pctT,pctP,pct};
}

function getCSS(v) { return getComputedStyle(document.documentElement).getPropertyValue(v).trim(); }

function renderChart(mode) {
  const canvas=document.getElementById('main-chart');
  if(chartInst){ chartInst.destroy(); chartInst=null; }

  let labels=[], dateKeys=[], title='';
  if(mode==='day'){
    const d=today();
    labels=['0〜4時','4〜8時','8〜12時','12〜16時','16〜20時','20〜24時'];
    dateKeys=labels; title='今日の時間帯別（分）';
    // 今日のlogを集計（時間帯分割は簡易：duration を均等6分割で表示）
    const todayLogs=db.logs.filter(l=>l.date===d);
    const datasets=db.subjects.map(s=>{
      const lg=todayLogs.find(l=>l.subjectId===s.id);
      const v=lg?Math.round(lg.duration/6):0;
      return {label:s.title,data:Array(6).fill(v),borderColor:s.color,backgroundColor:s.color+'33',borderWidth:2,tension:.3,fill:true,pointRadius:3};
    });
    buildChart(canvas,labels,datasets,title);
  } else if(mode==='week'){
    const days=getWeek();
    labels=['月','火','水','木','金','土','日'];
    const datasets=db.subjects.map(s=>({
      label:s.title,
      data:days.map(d=>{ const l=db.logs.find(x=>x.subjectId===s.id&&x.date===d); return l?l.duration:0; }),
      borderColor:s.color,backgroundColor:s.color+'33',borderWidth:2,tension:.3,fill:true,pointRadius:4
    }));
    buildChart(canvas,labels,datasets,'今週の日別学習時間（分）');
  } else {
    const days=getMonthDays();
    labels=days.map(d=>d.slice(8));
    const datasets=db.subjects.map(s=>({
      label:s.title,
      data:days.map(d=>{ const l=db.logs.find(x=>x.subjectId===s.id&&x.date===d); return l?l.duration:0; }),
      borderColor:s.color,backgroundColor:s.color+'22',borderWidth:1.5,tension:.3,fill:false,pointRadius:2
    }));
    buildChart(canvas,labels,datasets,'今月の日別学習時間（分）');
  }
}

function buildChart(canvas, labels, datasets, title) {
  const tc=getCSS('--text2'); const bc=getCSS('--border');
  chartInst=new Chart(canvas,{
    type:'line', data:{labels,datasets},
    options:{
      responsive:true,maintainAspectRatio:false,
      plugins:{
        legend:{labels:{color:tc,font:{size:11},boxWidth:12}},
        title:{display:!!title,text:title,color:tc,font:{size:11},padding:{bottom:8}}
      },
      scales:{
        x:{ticks:{color:getCSS('--text3'),font:{size:10}},grid:{color:bc}},
        y:{ticks:{color:getCSS('--text3'),font:{size:10}},grid:{color:bc},beginAtZero:true}
      }
    }
  });
}

function saveLog(src='manual', subjectId=null, duration=null) {
  const sid=subjectId||document.getElementById('l-sub').value;
  const date=document.getElementById('l-date')?.value||today();
  const dur=duration!==null?duration:parseInt(document.getElementById('l-time').value);
  const page=parseInt(document.getElementById('l-page')?.value)||0;
  const memo=document.getElementById('l-memo')?.value.trim()||'';

  if(!sid) return toast('科目を選択してください',true);
  if(isNaN(dur)||dur<0||dur>1440) return toast('時間は0〜1440分で入力してください',true);
  if(page<0) return toast('ページ数は0以上で入力してください',true);

  const idx=db.logs.findIndex(l=>l.subjectId===sid&&l.date===date);
  const rec={id:idx>=0?db.logs[idx].id:uid(),subjectId:sid,date,duration:dur,pageCount:page,comment:memo,src,savedAt:new Date().toISOString()};
  if(idx>=0) db.logs[idx]=rec; else db.logs.push(rec);
  save(db);
  if(src==='manual'){
    document.getElementById('l-time').value='';
    document.getElementById('l-page').value='';
    document.getElementById('l-memo').value='';
  }
  toast(idx>=0?'実績を更新しました':'実績を保存しました');
  renderLogPage();
}

// ========== TIMER OVERLAY ==========
const POMO_TOTAL = 25 * 60;
let timerState = 'IDLE'; // IDLE | RUNNING | PAUSED
let timerSec = POMO_TOTAL;
let timerElapsed = 0;    // 現セッションの経過秒（1ポモドーロ分）
let timerAccum  = 0;    // ★ 累積経過秒（複数セッション合計・勉強終了まで保持）
let timerInterval = null;
let timerSubjectId = null;

// ---------- overlay ----------
function overlayOpen() {
  const sel = document.getElementById('tov-subject');
  sel.innerHTML = db.subjects.length
    ? db.subjects.map(s=>`<option value="${s.id}">${s.title}</option>`).join('')
    : '<option value="">-- 先に科目を登録してください --</option>';
  // 科目は既に選択中なら維持
  if(timerSubjectId){
    for(let i=0;i<sel.options.length;i++){
      if(sel.options[i].value===timerSubjectId){ sel.selectedIndex=i; break; }
    }
  }
  document.getElementById('timer-overlay').classList.add('open');
  updateTimerUI();
}

function overlayClose() {
  document.getElementById('timer-overlay').classList.remove('open');
  updateDashTimerBadge();
}

// ---------- tick ----------
function _startTick() {
  clearInterval(timerInterval);
  timerInterval = setInterval(()=>{
    if(timerState !== 'RUNNING') return;
    timerSec--;
    timerElapsed++;
    timerAccum++;           // ★ 累積に1秒加算

    if(timerSec <= 0){
      // 25分1セッション完了 → 自動でリセットし次セッション待機
      clearInterval(timerInterval);
      timerState = 'IDLE';
      timerSec   = POMO_TOTAL;
      timerElapsed = 0;
      const accumMin = Math.floor(timerAccum / 60);
      toast(`🍅 25分完了！合計 ${accumMin}分 計測中。続けて開始できます`);
    }
    updateTimerUI();
    updateDashTimerBadge();
  }, 1000);
}

// ---------- controls ----------
function timerStart() {
  if(!db.subjects.length){ toast('先に科目を登録してください',true); return; }
  const sel = document.getElementById('tov-subject');
  // 初回 or セッション間の科目変更を許容
  if(timerAccum === 0){
    timerSubjectId = sel.value;
  }
  if(!timerSubjectId){ toast('科目を選択してください',true); return; }
  timerState   = 'RUNNING';
  timerSec     = POMO_TOTAL;
  timerElapsed = 0;
  _startTick();
  updateTimerUI();
}

function timerPause() {
  timerState = 'PAUSED';
  clearInterval(timerInterval);
  updateTimerUI();
}

function timerResume() {
  timerState = 'RUNNING';
  _startTick();
  updateTimerUI();
}

function timerEnd() {
  // ★ 累積秒 + 現セッションの未完了分を合算
  const totalSec  = timerAccum + (timerState !== 'IDLE' ? timerElapsed : 0);
  const totalMin  = Math.floor(totalSec / 60);

  if(totalMin < 1){ toast('1分以上計測してから終了してください',true); return; }

  clearInterval(timerInterval);
  const sid = timerSubjectId;

  // ★ 状態を完全リセット
  timerState    = 'IDLE';
  timerSec      = POMO_TOTAL;
  timerElapsed  = 0;
  timerAccum    = 0;
  timerSubjectId = null;

  overlayClose();

  // 実績入力ページへ遷移して値をセット
  document.querySelectorAll('.page').forEach(p=>p.classList.remove('active'));
  document.querySelectorAll('nav button:not(.theme-btn)').forEach(b=>b.classList.remove('active'));
  document.getElementById('page-log').classList.add('active');
  const logBtn = document.querySelector('nav button:nth-child(4)');
  if(logBtn) logBtn.classList.add('active');

  renderLogPage();

  // 科目・時間をセット
  const subSel = document.getElementById('l-sub');
  if(subSel && sid){
    for(let i=0;i<subSel.options.length;i++){
      if(subSel.options[i].value===sid){ subSel.selectedIndex=i; break; }
    }
  }
  const timeEl = document.getElementById('l-time');
  if(timeEl) timeEl.value = Math.min(totalMin, 1440);

  toast(`合計 ${totalMin}分 を実施時間に反映しました ✏️`);
  updateTimerUI();
}

// ---------- UI ----------
function updateTimerUI() {
  const m = Math.floor(timerSec/60).toString().padStart(2,'0');
  const s = (timerSec%60).toString().padStart(2,'0');
  document.getElementById('tov-display').textContent = `${m}:${s}`;

  // リング：現セッション進捗
  const circ = 2 * Math.PI * 108;
  const sessionSec = timerState !== 'IDLE' ? timerElapsed : 0;
  document.getElementById('tov-ring-fg').style.strokeDashoffset = circ * (1 - sessionSec / POMO_TOTAL);

  // ★ 経過表示：累積 + 現セッション
  const totalSec  = timerAccum + (timerState !== 'IDLE' ? timerElapsed : 0);
  const totalMin  = Math.floor(totalSec / 60);
  const totalSecR = totalSec % 60;
  const sessMin   = Math.floor(timerElapsed / 60);
  const sessSec   = timerElapsed % 60;

  let elapsedText = `合計 ${totalMin}分${totalSecR>0?totalSecR+'秒':''}`;
  if(timerAccum > 0 && timerState !== 'IDLE'){
    elapsedText += `（今回 ${sessMin}分${sessSec>0?sessSec+'秒':''}）`;
  }
  document.getElementById('tov-elapsed').textContent = elapsedText;

  const stateLabels = {
    IDLE: timerAccum>0 ? `${Math.floor(timerAccum/60)}分完了 — 続けて開始できます` : '科目を選んで開始してください',
    RUNNING: '集中中…',
    PAUSED:  '一時停止中'
  };
  document.getElementById('tov-state').textContent = stateLabels[timerState] || '';

  // 科目表示
  const selArea = document.getElementById('tov-sel-area');
  const subDisp = document.getElementById('tov-subject-display');
  if(timerState === 'IDLE' && timerAccum === 0){
    selArea.style.display=''; subDisp.style.display='none';
  } else {
    selArea.style.display='none'; subDisp.style.display='';
    const sub = db.subjects.find(s=>s.id===timerSubjectId);
    if(sub){
      document.getElementById('tov-sdot').style.background = sub.color;
      document.getElementById('tov-sname').textContent = sub.title;
    }
  }

  document.getElementById('tov-start-btn').style.display  = timerState==='IDLE'    ? '' : 'none';
  document.getElementById('tov-pause-btn').style.display  = timerState==='RUNNING' ? '' : 'none';
  document.getElementById('tov-resume-btn').style.display = timerState==='PAUSED'  ? '' : 'none';

  // 勉強終了：累積1分以上 or 計測中なら有効
  const totalMin2 = Math.floor((timerAccum+(timerState!=='IDLE'?timerElapsed:0))/60);
  const endBtn = document.getElementById('tov-end');
  endBtn.disabled      = totalMin2 < 1;
  endBtn.style.opacity = totalMin2 < 1 ? '0.4' : '1';
}

function updateDashTimerBadge() {
  const statusEl = document.getElementById('dash-timer-status');
  const stateEl  = document.getElementById('dash-timer-state');
  if(!statusEl) return;
  const m = Math.floor(timerSec/60).toString().padStart(2,'0');
  const s = (timerSec%60).toString().padStart(2,'0');
  statusEl.textContent = `${m}:${s}`;
  stateEl.textContent  = timerState==='RUNNING'?'計測中 🔴':timerState==='PAUSED'?'一時停止':'待機中';
  stateEl.style.color  = timerState==='RUNNING'?'#ef4444':'';
}

function importJSON(input) {
  const file=input.files[0]; if(!file) return;
  const reader=new FileReader();
  reader.onload=e=>{
    try{
      const p=JSON.parse(e.target.result);
      if(!validate(p)) throw new Error();
      db=p; save(db); toast('インポートしました'); renderBackup();
    }catch{ toast('不正なファイルです',true); }
    input.value='';
  };
  reader.readAsText(file);
}

    最終更新：${db.lastUpdated?new Date(db.lastUpdated).toLocaleString('ja-JP'):'—'}<br>
    バージョン：${db.version}`;
}
